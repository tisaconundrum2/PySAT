    Bands: 749, R909, R1109

    Parameters
    ----------
    data : ndarray
           (n,m,p) array

               (n,1) array of wavelengths that correspond to the p

       the processed ndarray
    '''
    wavelengths = [749, 909, 1109]
    return generic_func(data, wv_array, wavelengths, func = sp_funcs.curv_func)
